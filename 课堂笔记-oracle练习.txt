	1 创建表
		创建一个员工表t1，员工号id（整数），和姓名name（字符串）

		create table t1
		(
			id number,
			name varchar2(30)
		)


		char(30)   定长30个字符，不管写多少数据都是存30字符
			查询效率高一点，不用关注字符串不定长度，比如存储身份证18位，使用char合适
		varchar(30) 最大长度还是30，如果存储hello 5个字符，最终占用的5个字符的空间

	6 表的约束
		oracle支持5种约束
		非空   not null
		唯一	   unique
		检查   check
		主键   primary key
		外键   foreign key

		约束的定义
		create table 表名
		(
			列名1 类型1 [constraint 约束名] 约束类型以及参数,
			.....
		)

		create table student 
		(
			sid number constraint pk_student primary key, --学生Id主键约束
			sname varchar2(20) constraint nn_student_name not null,--学生姓名非空约束
			email varchar2(20) constraint un_student_email unique --学生邮件唯一约束
			constraint nn_student_email not null,	--同时邮件可再设非空，没有“，”
			age number constraint chk_student_age_min check(age > 10),	--学生年龄设置check约束
			gender varchar2(6) constraint chk_student_gender check(gender in ('男', '女')),
			deptno number constraint fk_student references dept (deptno) ON DELETE SET NULL
		)   

		查看某个表约束的定义
		select constraint_name, constraint_Type, search_condition  
			  from user_constraints where table_name='表名大写' 



2 视图

	创建方法
		create view 视图名 as 子查询

		跟另一种创建表的方法类似


	创一个视图，用来观看10号部门的员工信息
		create view v_emp10 as select * from emp where deptno = 10

		视图是没有存储数据，只是一个查询语句返回结果，每次我们查询视图的时候，都会去执行创建视图的查询语句


	删除视图
		drop view 视图名
		
		
3 序列

	每次插入数据的时候都要对一些主键id的数值字段进行自增，需要sql用户记住上一次到哪个值
	解决方法：让数据库自己记住序号到哪里
	create sequence 序列名

	使用方式

		序列名.currval   获取当前值
			select myseq.currval from dual
			       *
			第 1 行出现错误:
			ORA-08002: 序列 MYSEQ.CURRVAL 尚未在此会话中定义

			规定：执行currval不能早于第一次执行nextval，否则报错

		序列名.nextval   获取下一个值  

			同一条sql语句 ，nextval只取一次值


	删除序列
		drop sequence 序列名


	
4 索引
	原理就是对表的某一列或者多列进行排序，下次查询的时候，先查询索引，然后再到表中获取数据

	create index 索引名字 on 表名(列1,列2....)

	create index my_indx_deptno on emp(deptno)

	如何使用索引？

		不需要特定语句，查询时候oracle会自动根据条件来使用索引


		例子：
			对 emp表的 deptno , job 创建复合索引
			create index my_indx_deptno on emp(deptno,job)
			下次where条件 是   job = ??   ，没有自动使用索引 ,因为复合索引非第一列的排序
			还是无序的


	表有唯一约束的列（包括主键），oracle都会自动创建索引


	删除索引
		drop index 索引名字
